#include <iostream>
#include <stack>
#include <cstdlib>
#include "postfixCalculator.h"
using namespace std;

int main() {
    postfixCalculator p;
    p.push(1);
    p.push(2);
    p.push(3);
    p.push(4);
    p.push(5);
    p.add();
    p.add();
    p.add();
    p.add();
    cout << "Result of adding is: " << p.top() << endl;
    
    postfixCalculator p1;
    p1.push(1);
    p1.push(2);
    p1.push(3);
    p1.subtract();
    p1.subtract();
    cout << "Result of subtracting is: " << p1.top() << endl;

    postfixCalculator p2;
    p2.push(1);
    p2.push(2);
    p2.push(3);
    p2.multiply();
    p2.multiply();
    cout << "Result of multiplying is: " << p2.top() << endl;

    postfixCalculator p3;
    p3.push(4);
    p3.push(3);
    p3.push(2);
    p3.divide();
    p3.divide();
    cout << "Result of division is: " << p3.top() << endl;

    postfixCalculator p4;
    p4.push(-5);
    p4.push(2);
    p4.negate();
    cout << "Result of negation is: " << p4.top() << endl;

    return 0;
}
